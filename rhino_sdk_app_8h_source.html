203  /*
204  CRhinoApp::APP_STATE enum values mark points in the
205  sequence of the CRhinoApp being
206  constructed,
207  initializing,
208  closing,
209  exiting,
210  and destructed.
211  It is critical that the enum values be listed in chronological order.
212  */
213  enum APP_STATE : unsigned int
214  {
215  // app_state_ctor_* - CRhinoApp::CRhinoApp() {} status
216  app_state_ctor_zero = 0,
217  app_state_ctor_begin, // entered scope
218  app_state_ctor_end, // last line of scope
219  app_state_ctor_exited, // exited scope
220 
221  // app_state_InitInstance_* - CRhinoApp::InitInstance() {} status
222  app_state_InitInstance_begin, // entered scope
223  app_state_InitInstance_before_WinAppInit, // before call to CWinApp::InitInstance()
224  app_state_InitInstance_after_WinAppInit, // entering CRhinoApp::InitInstance() scope
225  app_state_InitInstance_end, // last line of scope
226  app_state_InitInstance_exited, // exited scope
227 
228  ///////////////////////////////////////////////////////////////////
229  //
230  // If m_rhino_app_state >= app_state_InitInstance_exited
231  // and m_rhino_app_state < app_state_OnClose_begin, then
232  // it means CRhinoApp::InitInstance() is finished and
233  // no WM_CLOSE message has been received by the main frame.
234  // This generally means Rhino is up and running.
235  // If more detailed app status information is needed, we
236  // can consider adding it. But keep in mind, this is Rhino
237  // app state and an app can have many docs and views
238  // in many states while it is running. This is not an appropriate
239  // place to report doc or view status because as soon as there
240  // are multiple docs or multiple views, there can be more
241  // than one answer to a doc or view question.
242  //
243  ///////////////////////////////////////////////////////////////////
244 
245  // app_state_OnClose_* - CMainFrame::OnClose() {} status
246  app_state_OnClose_begin, // entered scope
247  app_state_OnClose_before_MDIClose, // before call to CMDIFrameWnd::OnClose()
248  app_state_OnClose_UnsavedCancel, // CRhinoApp::SaveAllModified() - cancel requested
249  app_state_OnClose_after_MDIClose, // ordinary close - after call to CMDIFrameWnd::OnClose()
250  app_state_OnClose_end, // ordinary close - last line of scope
251  app_state_OnClose_Canceled, // canceled close - cleanup in progress
252  app_state_OnClose_exited, // exited scope
253 
254  // app_state_ExitInstance_* - CRhinoApp::InitInstance() {} status
255  app_state_ExitInstance_begin, // entered scope
256  app_state_ExitInstance_before_WinAppExit, // before call to CWinApp::InitInstance()
257  app_state_ExitInstance_after_WinAppExit, // entering CRhinoApp::InitInstance() scope
258  app_state_ExitInstance_end, // last line of scope
259  app_state_ExitInstance_exited, // exited scope
260 
261  app_state_dtor_begin, // entered scope
262  app_state_dtor_end, // last line of scope
263  app_state_dtor_exited // exited scope
264  };

621  /*
622  Description:
623  In general, you should use CRhinoApp::RunScript to run commands
624  unless you are trying to modify the Rhino model from a modeless
625  dialog. See the example below for how to handle the modeless
626  dialog situation.
627 
628  This is a low level tool that searches for a command (a instance
629  of a class derived from CRhinoCommand) with a matching command
630  name, if it finds a command, runs that command by calling the
631  virtual CRhinoCommand::RunCommand() member function.
632 
633  The command name can be prefaced with a combination of
634  underbar (_), hyphen (-), or right single quote ('). If an
635  underbar is present, then the command name search looks for
636  a matching english command name. If the underbar is not present,
637  the command name search looks for a matching localized command
638  name. If the hypen is present, then the command is run in script
639  mode; otherwise the command is run in interactive mode. The right
640  single quote is ignored by ExecuteCommand. It is used in scripts
641  to indicate that the text token must be treated as a command name
642  and is important to upstream text token parsing.
643 
644  Example:
645 
646  CMyCommand : pubic CRhinoCommand
647  {
648  public:
649  RunCommand(...);
650 
651  type1 m_my_info1;
652  type2 m_my_info2;
653  bool m_bOkToRunCommand;
654  };
655 
656  CMyCommand::CMyCommand()
657  {
658  m_bOkToRunCommand = false;
659  }
660 
661  CMyCommand theMyCommand;
662 
663  CMyCommand::RunCommand(...)
664  {
665  if (!m_bOkToRunCommand)
666  {
667  RhinoMessageBox(L"Use the MyDialog button to run this command");
668  return CRhinoCommand::failure;
669  }
670 
671  // the dialog has filled in m_my_info1 and m_my_info2.
672  // Use this as the "input" to the command and then
673  // do whatever you need to as if the command were
674  // an ordinary command.
675 
676  ...
677 
678  return CRhinoCommand::success;
679  }
680 
681  ...
682 
683  CMyDialog::OnButtonDown(...)
684  {
685  // fill in the parameters
686  theMyCommand.m_my_info1 = ...;
687  theMyCommand.m_my_info2 = ...;
688  theMyCommand.m_bOkToRunCommand = true;
689  RhinoApp().ExectueCommand(theMyCommand.LocalCommandName());
690  theMyCommand.m_bOkToRunCommand = false;
691  }
692 
693  Returns:
694  Command return code indicating if the command worked, failed,
695  or was canceled by the user.
